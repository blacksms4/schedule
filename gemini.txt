import { initializeApp } from 'firebase/app';
import { getAuth, signInAnonymously, onAuthStateChanged } from 'firebase/auth';
import { getFirestore, doc, setDoc, onSnapshot } from 'firebase/firestore';

const app = initializeApp(firebaseConfig);
const auth = getAuth(app);
const db = getFirestore(app);

const holidays = { "2026-06-06": "현충일", "2026-07-17": "제헌절", "2026-08-15": "광복절" };

export default function App() {
  const [user, setUser] = useState(null);
  const [players, setPlayers] = useState([]);
  const [ladderLines, setLadderLines] = useState([]);
  const [finalResults, setFinalResults] = useState([]);
  const [currentYear, setCurrentYear] = useState(2026);
  const [currentMonth, setCurrentMonth] = useState(5);
  const [assignments, setAssignments] = useState({});
  const [activeTab, setActiveTab] = useState('calendar');
  const [playerInput, setPlayerInput] = useState('');
  const [workerCount, setWorkerCount] = useState(2);
  const canvasRef = useRef(null);

  useEffect(() => {
    onAuthStateChanged(auth, (user) => { user ? setUser(user) : signInAnonymously(auth); });
  }, []);

  useEffect(() => {
    if (!user) return;
    const docRef = doc(db, 'data', 'assignments');
    return onSnapshot(docRef, (doc) => { if (doc.exists()) setAssignments(doc.data()); });
  }, [user]);

  const saveAssignment = async (updated) => {
    await setDoc(doc(db, 'data', 'assignments'), updated, { merge: true });
  };

  const changeMonth = (offset) => {
    let newMonth = currentMonth + offset;
    let newYear = currentYear;
    if (newMonth < 0) { newMonth = 11; newYear -= 1; }
    else if (newMonth > 11) { newMonth = 0; newYear += 1; }
    setCurrentMonth(newMonth);
    setCurrentYear(newYear);
  };

  const drawLadder = () => {
    if (players.length < 2) return alert("최소 2명이 필요합니다.");
    const canvas = canvasRef.current;
    const ctx = canvas.getContext('2d');
    ctx.clearRect(0, 0, canvas.width, canvas.height);
    const colWidth = canvas.width / (players.length + 1);
    
    players.forEach((p, i) => {
      let x = (i + 1) * colWidth;
      ctx.strokeStyle = '#94a3b8'; ctx.lineWidth = 4;
      ctx.beginPath(); ctx.moveTo(x, 40); ctx.lineTo(x, 260); ctx.stroke();
      ctx.fillStyle = '#1e293b'; ctx.textAlign = 'center'; ctx.font = '12px Arial'; ctx.fillText(p, x, 30);
    });

    const lines = [];
    for (let row = 0; row < 10; row++) {
        let y = 70 + row * 18;
        for (let col = 1; col < players.length; col++) {
            if (Math.random() > 0.4) {
                if (!lines.find(l => l.y === y && (l.col === col - 1 || l.col === col + 1))) {
                    lines.push({ col, y });
                    ctx.strokeStyle = '#3b82f6'; ctx.lineWidth = 3;
                    ctx.beginPath(); ctx.moveTo(col * colWidth, y); ctx.lineTo((col + 1) * colWidth, y); ctx.stroke();
                }
            }
        }
    }
    setLadderLines(lines);
    setFinalResults([]);
  };

  const handleCanvasClick = (e) => {
    const canvas = canvasRef.current;
    const rect = canvas.getBoundingClientRect();
    const colWidth = canvas.width / (players.length + 1);
    const x = e.clientX - rect.left;
    const startCol = Math.max(1, Math.min(players.length, Math.round(x / colWidth)));
    
    if (finalResults.some(r => r.name === players[startCol - 1])) return;
    
    const ctx = canvas.getContext('2d');
    let curCol = startCol;
    let curY = 40;
    const sortedLines = [...ladderLines].sort((a, b) => a.y - b.y);
    
    ctx.strokeStyle = '#ef4444'; ctx.lineWidth = 3;
    ctx.beginPath(); ctx.moveTo(startCol * colWidth, curY);

    for (let line of sortedLines) {
        if (line.y > curY) {
            if (line.col === curCol) {
                ctx.lineTo(curCol * colWidth, line.y);
                curCol += 1;
                ctx.lineTo(curCol * colWidth, line.y);
                curY = line.y;
            } else if (line.col === curCol - 1) {
                ctx.lineTo(curCol * colWidth, line.y);
                curCol -= 1;
                ctx.lineTo(curCol * colWidth, line.y);
                curY = line.y;
            }
        }
    }
    ctx.lineTo(curCol * colWidth, 260);
    ctx.stroke();
    
    const result = { name: players[startCol - 1], target: curCol };
    setFinalResults(prev => [...prev, result]);
  };

  const assignToCalendar = () => {
    if (finalResults.length < players.length) return alert("모든 플레이어가 사다리를 타야 합니다.");
    const newAssignments = { ...assignments };
    const daysInMonth = new Date(currentYear, currentMonth + 1, 0).getDate();
    const workerList = finalResults.sort((a,b)=>a.target-b.target).map(r => r.name);
    let playerIdx = 0;
    for (let i = 1; i <= daysInMonth; i++) {
      let d = new Date(currentYear, currentMonth, i);
      let dateKey = `${currentYear}-${(currentMonth + 1).toString().padStart(2, '0')}-${i.toString().padStart(2, '0')}`;
      if (d.getDay() === 0 || d.getDay() === 6 || holidays[dateKey]) {
        let dayWorkers = [];
        for (let w = 0; w < workerCount; w++) {
            dayWorkers.push(workerList[playerIdx % workerList.length]);
            playerIdx++;
        }
        newAssignments[dateKey] = dayWorkers.join(', ');
      } else {
        newAssignments[dateKey] = '';
      }
    }
    setAssignments(newAssignments);
    saveAssignment(newAssignments);
    alert(`${currentYear}년 ${currentMonth + 1}월 근무표가 반영되었습니다.`);
  };

  return (
    <div className="p-4 max-w-lg mx-auto">
      <h1 className="text-xl font-bold mb-4 text-center">휴일 근무 관리</h1>
      
      <div className="flex gap-2 mb-4">
        <button onClick={() => setActiveTab('calendar')} className={`flex-1 py-2 ${activeTab === 'calendar' ? 'bg-blue-600 text-white' : 'bg-gray-200'}`}>달력</button>
        <button onClick={() => setActiveTab('ladder')} className={`flex-1 py-2 ${activeTab === 'ladder' ? 'bg-blue-600 text-white' : 'bg-gray-200'}`}>사다리</button>
      </div>

      {activeTab === 'calendar' ? (
        <div>
          <div className="flex justify-between items-center mb-2">
            <button onClick={() => changeMonth(-1)} className="bg-gray-300 px-3 py-1 rounded">이전 달</button>
            <h2 className="font-bold">{currentYear}년 {currentMonth + 1}월</h2>
            <button onClick={() => changeMonth(1)} className="bg-gray-300 px-3 py-1 rounded">다음 달</button>
          </div>
          <div className="grid grid-cols-7 border text-center">
            {['일','월','화','수','목','금','토'].map((d, i) => <div key={d} className={`font-bold text-xs p-1 ${i === 0 ? 'text-red-500' : i === 6 ? 'text-blue-500' : ''}`}>{d}</div>)}
            {[...Array(new Date(currentYear, currentMonth, 1).getDay())].map((_, i) => <div key={`e-${i}`} className="h-16 border" />)}
            {[...Array(new Date(currentYear, currentMonth + 1, 0).getDate())].map((_, i) => {
              const d = new Date(currentYear, currentMonth, i + 1);
              const key = `${currentYear}-${(currentMonth + 1).toString().padStart(2, '0')}-${(i + 1).toString().padStart(2, '0')}`;
              const isWeekend = d.getDay() === 0 || d.getDay() === 6 || holidays[key];
              return (
                <div key={i} className={`p-1 border h-16 ${isWeekend ? 'bg-red-50' : ''}`}>
                  <div className={`font-bold text-xs ${d.getDay() === 0 ? 'text-red-500' : d.getDay() === 6 ? 'text-blue-500' : ''}`}>{i + 1}</div>
                  <div className="text-[9px] truncate">{assignments[key] || ''}</div>
                </div>
              );
            })}
          </div>
        </div>
      ) : (
        <div className="space-y-2">
          <div className="text-center font-bold bg-blue-50 p-2 rounded">현재 선택된 달: {currentYear}년 {currentMonth + 1}월</div>
          <input className="border w-full p-1" value={playerInput} onChange={e => setPlayerInput(e.target.value)} placeholder="이름 입력" />
          <button className="w-full bg-blue-500 text-white" onClick={() => { if(playerInput) { setPlayers([...players, playerInput]); setPlayerInput(''); } }}>추가</button>
          <div className="flex flex-wrap gap-1">{players.map((p, i) => <span key={i} className="bg-gray-200 px-2 rounded-full text-sm">{p}</span>)}</div>
          <button className="w-full bg-green-500 text-white" onClick={drawLadder}>사다리 생성</button>
          <canvas ref={canvasRef} width={400} height={280} onClick={handleCanvasClick} className="border w-full cursor-pointer bg-white" />
          <div className="text-xs bg-yellow-100 p-2">결과: {finalResults.map(r => `${r.name} > ${r.target}번`).join(', ')}</div>
          <button className="w-full bg-indigo-600 text-white" onClick={assignToCalendar}>선택 달 근무표 반영</button>
        </div>
      )}
    </div>
  );
}
